import cv2
import numpy as np
import os
import shutil
import re
import gc
from pathlib import Path
from PIL import Image
import pillow_heif
import fitz  # PyMuPDF

pillow_heif.register_heif_opener()

class StudioProProcessor:
    def __init__(self):
        self.base_path = Path(__file__).parent
        self.backup_path = self.base_path / "originals"
        self.backup_path.mkdir(exist_ok=True)
        self.valid_exts = {'.jpg', '.jpeg', '.png', '.heic', '.heif', '.webp', '.pdf'}

    def get_info(self, filename):
        name, ext = os.path.splitext(filename)
        match = re.search(r'^(.*)_e(\d+)$', name)
        if match:
            return match.group(1), int(match.group(2)), ext
        return name, 0, ext

    def adjust_lighting(self, img, mode):
        if img.dtype != np.uint8:
            img = cv2.convertScaleAbs(img)
            
        if mode == '4': # DARKMODE FIX
            lab = cv2.cvtColor(img, cv2.COLOR_BGR2LAB)
            l_channel, _, _ = cv2.split(lab)
            if np.mean(l_channel) < 100:
                inverted = cv2.bitwise_not(img)
                return cv2.convertScaleAbs(inverted, alpha=1.2, beta=-10)
            return img

        lab = cv2.cvtColor(img, cv2.COLOR_BGR2LAB)
        l, a, b = cv2.split(lab)
        
        if mode == '1': # Fair
            clahe = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(8, 8))
            l = clahe.apply(l)
            img = cv2.cvtColor(cv2.merge((l, a, b)), cv2.COLOR_LAB2BGR)
            return cv2.convertScaleAbs(img, alpha=1.1, beta=5)
        elif mode == '2': # Light
            l = cv2.add(l, 25)
        elif mode == '3': # Dark
            l = cv2.subtract(l, 25)
            
        return cv2.cvtColor(cv2.merge((l, a, b)), cv2.COLOR_LAB2BGR)

    def undo(self):
        files = [f for f in self.base_path.iterdir() if f.suffix.lower() in self.valid_exts]
        undone_count = 0
        for file_path in files:
            base, version, ext = self.get_info(file_path.name)
            if version == 0: continue 
            prev_v = version - 1
            search = f"{base}_e{prev_v}{ext}" if prev_v > 0 else f"{base}{ext}"
            backup_source = self.backup_path / search
            if backup_source.exists():
                file_path.unlink()
                shutil.copy(str(backup_source), str(self.base_path / search))
                undone_count += 1
        print(f"⏪ Reverted {undone_count} files.")
        gc.collect()

    def run(self, mode):
        files = [f for f in self.base_path.iterdir() if f.suffix.lower() in self.valid_exts]
        
        for file_path in files:
            if file_path.name == Path(__file__).name: continue
            
            try:
                shutil.copy(str(file_path), str(self.backup_path / file_path.name))
                base, version, ext = self.get_info(file_path.name)
                new_name = f"{base}_e{version + 1}{ext}"
                
                if ext.lower() == '.pdf':
                    doc = fitz.open(file_path)
                    new_pdf = fitz.open() # Create new empty PDF
                    
                    for i in range(len(doc)):
                        page = doc[i]
                        pix = page.get_pixmap(dpi=200)
                        img_data = np.frombuffer(pix.samples, dtype=np.uint8).reshape(pix.h, pix.w, pix.n)
                        cv_img = cv2.cvtColor(img_data, cv2.COLOR_RGB2BGR) if pix.n == 3 else img_data
                        
                        # Process frame
                        res = self.adjust_lighting(cv_img, mode)
                        
                        # Convert processed frame to bytes for PDF insertion
                        _, img_encoded = cv2.imencode(".jpg", res, [int(cv2.IMWRITE_JPEG_QUALITY), 90])
                        img_bytes = img_encoded.tobytes()
                        
                        # Add to new PDF
                        img_doc = fitz.open("jpg", img_bytes)
                        pdf_bytes = img_doc.convert_to_pdf()
                        img_doc.close()
                        
                        page_rect = new_pdf.new_page(width=page.rect.width, height=page.rect.height)
                        page_rect.show_pdf_page(page_rect.rect, fitz.open("pdf", pdf_bytes), 0)
                        
                    new_pdf.save(str(self.base_path / new_name))
                    new_pdf.close()
                    doc.close()
                else:
                    with Image.open(file_path) as pil_img:
                        cv_img = cv2.cvtColor(np.array(pil_img.convert('RGB')), cv2.COLOR_RGB2BGR)
                    enhanced = self.adjust_lighting(cv_img, mode)
                    cv2.imwrite(str(self.base_path / new_name), enhanced)
                
                file_path.unlink()
                print(f"✅ Processed: {new_name}")
                gc.collect() 

            except Exception as e:
                print(f"❌ Error on {file_path.name}: {e}")

if __name__ == "__main__":
    proc = StudioProProcessor()
    while True:
        print("\n--- STUDIO PRO (PDF SUPPORTED) ---")
        print("1. Fair | 2. Light | 3. Dark | 4. FIX DARKMODE (Printer-Ready)")
        print("5. UNDO | 6. EXIT")
        c = input("Choice: ")
        if c in ['1', '2', '3', '4']: 
            proc.run(c)
        elif c == '5': 
            proc.undo()
        elif c == '6': 
            break
